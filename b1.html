JavaScript Syntax Examples

document.getElementById()

// Mengakses elemen dengan ID tertentu
const element = document.getElementById("demo");
element.innerHTML = "Teks diubah!";
element.style.color = "red";

// Contoh lainnya:
// document.getElementById("myImage").src = "new_image.jpg";
// document.getElementById("myDiv").style.display = "none";
Penjelasan: Method document.getElementById() digunakan untuk mengakses elemen HTML berdasarkan ID-nya. Setelah elemen diakses, kita bisa memanipulasi konten, style, atribut, dan properti lainnya.

var, const, let

// Deklarasi variabel
var nama = "Afif"; // Function-scoped
let umur = 25; // Block-scoped
const pi = 3.14; // Nilai konstan

// Perbedaan scope
if (true) {
  var a = 10;
  let b = 20;
}

console.log(a); // 10 (dapat diakses)
console.log(b); // Error: b is not defined
Penjelasan:
  • var: Function-scoped, bisa di-redeclare
  • let: Block-scoped, tidak bisa di-redeclare
  • const: Block-scoped, nilai tetap (tidak bisa reassign)
Gunakan const secara default, let jika perlu reassign, dan hindari var.

if Statement

const nilai = 85;

if (nilai >= 90) {
  console.log("Grade: A");
} else if (nilai >= 80) {
  console.log("Grade: B");
} else if (nilai >= 70) {
  console.log("Grade: C");
} else {
  console.log("Grade: D");
}
Penjelasan: Statement if digunakan untuk mengeksekusi blok kode jika kondisi yang diberikan bernilai truthy. else if digunakan untuk kondisi alternatif, dan else untuk semua kondisi lainnya.

switch Statement

const hari = "Senin";
let aktivitas;

switch (hari) {
  case "Senin":
    aktivitas = "Meeting tim";
    break;
  case "Rabu":
    aktivitas = "Presentasi klien";
    break;
  case "Jumat":
    aktivitas = "Review mingguan";
    break;
  default:
    aktivitas = "Coding";
}
console.log(`Hari ${hari}: ${aktivitas}`);
Penjelasan: Statement switch mengevaluasi ekspresi dan mengeksekusi kode yang sesuai dengan case yang cocok. break penting untuk menghentikan eksekusi ke case berikutnya. default dijalankan jika tidak ada case yang cocok.

for Loop

// Loop dasar
for (let i = 0; i < 5; i++) {
  console.log(`Iterasi ke-${i}`);
}

// Loop melalui array
const buah = ["Apel", "Mangga", "Jeruk"];
for (let i = 0; i < buah.length; i++) {
  console.log(buah[i]);
}

// for...of loop
for (const b of buah) {
  console.log(b);
}
Penjelasan: Loop for memiliki tiga bagian: inisialisasi, kondisi, dan increment/decrement. for...of digunakan untuk iterasi elemen array. Ada juga for...in untuk iterasi properti object.

Function

// Deklarasi fungsi
function sapa(nama) {
  return `Halo, ${nama}!`;
}

// Ekspresi fungsi
const tambah = function(a, b) {
  return a + b;
};

// Arrow function
const kali = (a, b) => a * b;

console.log(sapa("Budi"));
console.log(tambah(5, 3));
console.log(kali(4, 6));
Penjelasan: Fungsi adalah blok kode yang dapat dipanggil berulang kali. Ada tiga cara mendefinisikan fungsi:
  1. Deklarasi fungsi: function name() {}
  2. Ekspresi fungsi: const name = function() {}
  3. Arrow function: const name = () => {}

return Statement

function cekGenap(angka) {
  if (angka % 2 === 0) {
    return true;
  }
  return false;
}

function hitungLuas(panjang, lebar) {
  return panjang * lebar;
}

console.log(cekGenap(4)); // true
console.log(cekGenap(7)); // false
console.log(hitungLuas(5, 4)); // 20
Penjelasan: Statement return menghentikan eksekusi fungsi dan mengembalikan nilai ke pemanggil. Jika tidak ada return, fungsi mengembalikan undefined.

try...catch

try {
  // Kode yang mungkin menghasilkan error
  const hasil = 10 / 0;
  if (!isFinite(hasil)) {
    throw new Error("Pembagian dengan nol!");
  }
  console.log(hasil);
} catch (error) {
  // Menangani error
  console.error("Terjadi error:", error.message);
} finally {
  // Selalu dijalankan
  console.log("Proses selesai");
}
Penjelasan: Blok try...catch menangani error runtime. Kode dalam try dijalankan, jika terjadi error, eksekusi berpindah ke catch. Blok finally selalu dijalankan, baik terjadi error maupun tidak. throw digunakan untuk melempar error kustom.

Object Property Access

const person = {
  name: "Afif",
  age: 25,
  "hobbies": ["coding", "gaming"]
};

// Dot notation
console.log(person.name); // Afif

// Bracket notation
console.log(person["age"]); // 25

// Expression
const prop = "hobbies";
console.log(person[prop]); // ["coding", "gaming"]

// Menambah properti
person.job = "Developer";

// Menghapus properti
delete person.age;
Penjelasan: Ada tiga cara mengakses properti object:
  1. Dot notation: object.property (untuk nama properti valid)
  2. Bracket notation: object["property"] (untuk nama dengan spasi/simbol)
  3. Expression: object[expression] (ketika properti disimpan dalam variabel)
Properti dapat ditambah dengan assignment dan dihapus dengan delete.

Object Methods & Constructor

// Object dengan method
const person = {
  firstName: "Afif",
  lastName: "Maulana",
  fullName: function() {
    return `${this.firstName} ${this.lastName}`;
  }
};

console.log(person.fullName());

// Constructor function
function Person(first, last, age) {
  this.firstName = first;
  this.lastName = last;
  this.age = age;
  this.getInfo = function() {
    return `${this.firstName} ${this.lastName}, ${age} tahun`;
  };
}

// Membuat instance object
const person1 = new Person("Budi", "Santoso", 30);
console.log(person1.getInfo());
Penjelasan:
  • Object dapat memiliki method (fungsi sebagai properti)
  • this merujuk pada object yang memiliki method
  • Constructor function (dengan awalan kapital) digunakan untuk membuat banyak object dengan struktur serupa
  • new keyword digunakan untuk membuat instance baru
